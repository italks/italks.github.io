def eniac_trajectory(v0, angle):
    """模拟炮弹轨迹"""
    import math
    
    # 初始参数
    g = 9.8  # 重力加速度
    rad = math.radians(angle)
    
    # 计算轨迹
    for t in range(0, 100):
        x = v0 * math.cos(rad) * t
        y = v0 * math.sin(rad) * t - 0.5 * g * t**2
        
        if y < 0:
            break
            
        print(f"时间:{t}s 位置:({x:.1f}, {y:.1f})")

# 运行
eniac_trajectory(500, 45)
